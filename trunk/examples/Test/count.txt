/****************************
 * Block of Semantic Domain *
 ****************************/
semantic domain

    /*********************
     * List of Operators *
     *********************/
    op infix    (10, left) +: int, int -> int;
    
    function    f: int   -> int;
    
/***********************
 * Block of Attributes *
 ***********************/
attributes
    longi : syn <int> of {S,T,T2};
    
/******************
 * Block of Rules *
 ******************/
rules
    S ::= T
            compute
                S[0].longi = T[0].longi;
            end;

    // P1
    T ::= '0' T2 T T
            compute             
                T[0].longi = T2[0].longi + T[1].longi + T[2].longi;
            end;
            
    T2 ::= '1'
            compute
                T2[0].longi = 1;
            end;
            
    T ::= '0'
            compute
                T[0].longi = 0;
            end;
