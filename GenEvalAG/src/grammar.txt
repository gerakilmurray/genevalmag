/****************************
 * Block of Semantic Domain *
 ****************************/
semantic domain
	/*****************
	 * List of Sorts *
	 *****************/
	sort int;
	sort string, character; // lala
	sort pepe;
	sort adrian;

	/*********************
	 * List of Operators *
	 *********************/
	op infix	(5, left)		+  : int, int	-> int;
	op infix	(50,non-assoc)	*  : int, int	-> int;
	op 			(99,_)			sqr: int		-> int;
	op postfix					&  : int		-> int;	
	op prefix	(_,right)		%  : string		-> string;
	op 			(8,non-assoc)	%  : string		-> adrian;
	op infix	(5,left)		+  : int, int	-> string;
	op prefix (_,left) + : int -> int;
		
	/*********************
	 * List of Functions *
	 *********************/
	function	f: int, int, character -> string;
	function	g: string	-> int;
	
/***********************
 * Block of Attributes *
 ***********************/
attributes
	valor			: syn	<int>		of all;
	lexema, termino	: syn	<string>	of all - {E};
	lexx			: syn	<string>	of all - {T};
	type			: inh	<string>	of all;
	grade			:		<int>		of {E};
	
/******************
 * Block of Rules *
 ******************/
rules
	E ::=	E '+' T
				compute
					E[0].valor = T[0].termino * +(%T[0].valor& + T[0].valor);
				end
		| 	T
			  	compute
			  		E[0].valor =  f (T[0].valor& & & & &);
	  			end
		;

	T ::=	id '*' id
				compute
					T[0].valor = id[0].valor * id[0].valor;
					T[0].type = "id";
				end
		;

	id ::=	'1' 
				compute
					id[0].valor = 1; 
					id[0].type = "int"; 
				end
	     |  '0'
				compute
					id[0].valor = '0' + '1'; 
					id[0].type = "int"; 
				end
		;
