\frame{
\frametitle{\textquestiondown Qué es \maggen?}
    \begin{block}{}
        \maggen\ es una herramienta generadora de evaluadores estáticos para la familia MAG. 
    \end{block}

    \begin{center}
        {\huge \textbf{mag} + \textbf{Gen}}\\
        \textit{Multi-plans Attribute Grammar} - \textit{Generator}
    \end{center}
    \pause

    Características destacables:
    \begin{itemize}
        \item Funciona sobre la familia MAG.
        \item La construcción de planes mantiene la propiedad de ``unicidad de planes''.
        \item Construcción de planes solo para reglas alcanzables.
        \item Generación de Evaluadores estáticos.
        \item La generación de código produce una representación compacta de cada estructura. 
    \end{itemize}

% \item Tanto el algoritmo de construcción de planes, como el de secuencia de visitas mantienen propiedades de compatibilidad y consistencia con los grafos de dependencia aumentados (ADP).
}

\subsection{Motivación}

\frame{
    \frametitle{Motivación}
    
    \begin{block}{}
        Los lenguajes juegan un rol muy importante en muchas disciplinas de la computación. Siempre se buscaron \textit{mecanismos} para describirlos y manipularlos.
    \end{block}
    
    \begin{block}{}
        Las \textbf{Gramáticas de Atributos} (GA), introducidas por D. Knuth, se han utilizado en herramientas de procesamiento de lenguajes y para especificar la semántica de lenguajes de programación.
    \end{block}
    
    \begin{block}{}
        En 1998, Wuu Yang caracteriza una nueva familia denominada \textbf{Gramática de Atributos Multiplanes}.
    \end{block}
    \pause
    
    \begin{center}
        {\large\textbf{Al momento no existían herramientas basadas en esta familia de Gramáticas de Atributos.}}
    \end{center} 
}

\subsection*{Notación}

\frame{
    \frametitle{Notación}
    \begin{block}{Gramática de atributos (Informal)}
        En una \textbf{Gramática de Atributos} (GA), se relaciona cada símbolo de una \textit{Gramática Libre de Contexto} con un conjunto de atributos. Cada regla o producción tiene asociado un \textit{conjunto de reglas semánticas}, denominadas también \textit{ecuaciones}.
    \end{block}

    \textbf{Instancia:} Ocurrencia de un símbolo de la gramática asociado a un atributo. 
}

\subsection{Gramáticas de Atributos Multiplanes (MAG)}
\frame{
    \frametitle{Gramáticas de Atributos Multiplanes}

    \begin{block}{Dependencias directas de una producción}
        Se denota como \textbf{DP(p)}	
        $$
          DP(p) = \{(X_{i}.a, X_{j}.b) | X_{i}.a \rightarrow Y_{j}.b \in R^{p} \}
        $$
    \end{block}
    
    Dado un símbolo X de una gramática de atributos

    $$\textbf{Down(X)} =  \{(a,b) | a \rightarrow b \}\ con\ a,b \in A(X)\}$$
}

\frame{
    \frametitle{Gramáticas de Atributos Multiplanes (cont.)}

    \begin{block}{Conjunto de Dependencias Aumentadas}
        Sea $q$ una producción de la forma $X_{0}\rightarrow \alpha_{0} X_{1} \alpha_{1} X_{2} \dots X_{k} \alpha_{k}$. Sea $p_{i}$ una producción cuya parte izquierda es $X_{i}$ ($1\leqslant i \leqslant k$). 
        $$
        ADP (q | p_{1}, p_{2}, \dots, p_{k}) = DP(q) \bigcup\limits_{k}^{i=1}{DGC_{X_{i}}} (p_{i})
        $$
        se denota como $\textbf{ADP (q $|$ p}_{\textbf{1}}\textbf{, p}_{\textbf{2}}\textbf{, \dots , p}_{\textbf{k}}\textbf{)} $
    \end{block}
    
    $\textbf{DCG}_{\textbf{X}}\textbf{(p)}$ contiene las dependencias, entre las instancias de la gramática, para el símbolo \texttt{X}, acotando el análisis para la producción \textit{p} y los posibles contextos inferiores.
    $$
    \bigcup\limits_{\textit{todo p}}{DCG_{X} (p) = Down (X)}
    $$
}

\frame{
    \frametitle{Gramáticas de Atributos Multiplanes (cont.)}

    El conjunto de todas las posibles dependencias aumentadas para una producción \textbtt{q} se define como:
    $$
    SADP(q) = \bigcup\limits_{q\in P}{ADP (q | p_{1}, p_{2}, \dots, p_{k})} 
    $$

    \begin{exampleblock}{Gramática de Atributos Multiplanes (MAG)}
        Una gramática \textit{G} es una \textit{Gramática de Atributos Multiplanes} si y sólo si 
        $$
        \forall q: q \in P: (\forall g: g\ es\ un\ grafo\ de\ q \wedge g \in SADP(q): g\ es\ no\ circular) 
        $$
    \end{exampleblock}
}

\subsection{Evaluador estático}

\frame{
    \frametitle{Evaluador estático}
    
    \begin{block}{Funcionamiento}
        \begin{itemize}
            \item Considera todos los posibles árboles sintácticos de la gramática.
            \item Calcula las dependencias entre todas las instancias de los atributos.
        \end{itemize}
    \end{block}

    \begin{block}{Contexto de una producción $p$}
        Compuesto por 3 tipos de dependencias:
        \begin{itemize}
            \item Directas obtenidas por las ecuaciones de $p$.
            \item Impuestas por el contexto superior.
            \item Impuestas en el contexto inferior.
        \end{itemize}
    \end{block}

%     Instancias diferentes de una producción $p$ tendrán las mismas dependencias directas, pero podrán tener diferentes dependencias impuestas por los contextos inferiores y superiores. 
}

\subsection*{Chequeos}

\frame{
    \frametitle{\maggen: Chequeos}

    Sobre la MAG de entrada se verifican:
    \begin{itemize}
        \item Precedencia y Asociatividad de operadores dentro de expresiones.

	\item Alcanzabilidad de símbolos y reglas.

        \item Consistencia de bloque de atributos y bloque de reglas.

        \item Gramática bien definida
        \pause
        \begin{itemize}
            \item Atributos sintetizados del símbolo de la parte izquierda.

            \item Atributos heredados de símbolos de la parte derecha.

            \item Ecuaciones sólo de los atributos de los símbolos de la regla.

            \item Inexistencia de ecuaciones para atributos heredados del símbolo de la izquierda y para atributos sintetizados de símbolos de la parte derecha.

            \item Múltiples ecuaciones para un atributo.

            \item Consistencia de los índices de las instancias que aparecen en las ecuaciones.

            \item Condiciones de una gramática extendida.
        \end{itemize}
    \end{itemize}
}

\subsection*{Generación de grafos}

\frame{
    \frametitle{\maggen: Generación de grafos}
    
    Luego de los chequeos semánticos sobre la gramática de entrada, se construyen los siguientes grafos:

    \begin{itemize}
	\item Grafos \textbf{DP}%: Dependencias directas de cada producción.

	\item Grafos \textbf{Down}%: Dependencias transitivas entre atributos de cada símbolo.

	\item Grafos \textbf{DCG}%: Dependencias transitivas entre atributos de cada símbolos acotado a una producción.

	\item Grafos \textbf{ADP}%: Grafos de dependencias transitivas aumentadas de cada producción.
    \end{itemize}
    \pause

    \begin{block}{}
	La construcción de los grafos permite realizar chequeos de ciclicidad y sirven de base (Grafos ADP) para la construcción de los planes de evaluación.
    \end{block}
}

\subsection*{Ciclicidad}

\begin{frame}[fragile]
    \frametitle{Ciclicidad}

    \vspace{-0.5cm}
    \begin{block}{}
	\begin{itemize}
	    \item Es realizado sobre los grafos ADP.

	    \item Basado en el algoritmo de búsqueda en profundidad (Depth-first search) de \textbf{Boost} combinado con un ``visitador'' especializado.

	    \item El subgrafo cíclico, es mostrado al usuario en caso de detección de ciclo.
	\end{itemize}
    \end{block}
    \pause

La salida de \maggen\ cuando se detecta ciclicidad, es:

\begin{lstlisting}[backgroundcolor=\color{white}, basicstyle=\footnotesize, language=] 
   * Parsing grammar ---------- [  OK  ]
   * Generate graphs ---------- [  OK  ]
   * Build plans -------------- [ ABORT ]

   ERROR: One o more graph ADP has an cycle in its dependencies. Look the folder ./out_maggen/graphs/CYCLIC_graphs/ for more details.
\end{lstlisting}
\end{frame}

\subsection*{Plan de evaluación}

\frame{
    \frametitle{\maggen: Plan de evaluación}

    Los grafos ADP son la entrada para la construcción de \textbf{planes de evaluación}.

    \begin{block}{Unicidad}
    \begin{itemize}
     \item Se mantiene un repositorio indexado para impedir elementos repetidos.
     \item Cada índice de plan referencia a una posición de un \textit{vector de planes únicos}.
     \item Optimización notable tanto en tiempo como en espacio.
    \end{itemize}
    \end{block}
    \pause
        
    \begin{block}{Algoritmo \textbtt{compute\_plans}}
	\begin{itemize}
	    \item Propuesto por Wuu Yang en \cite{wuu-yang1}.
	    \item Generación de planes y planes proyectados de evaluación.
	\end{itemize}
    \end{block}
}

\frame{
    \frametitle{\maggen: Plan de evaluación (cont.)}

    \begin{block}{Plan de evaluación}
        Lista de números de ecuaciones a computar, dirigiendo la evaluación de izquierda a derecha.
    \end{block}
    \pause
    
    \begin{block}{Plan de evaluación proyectado}
        Plan de evaluación considerando sólo los números de ecuaciones para los atributos de un símbolo particular.
    \end{block}
    \pause

    \begin{center}
        ``Los planes proyectados juegan un rol restrictivo sobre el orden de evaluación impuesto al contexto inferior.''
    \end{center}
}

\subsection*{Secuencias de visita}

\frame{
  \frametitle{\maggen: Secuencias de visita}

  \begin{block}{}
    \begin{itemize}
      \item Traducción de planes de evaluación a secuencias de visita: \textbtt{compute}, \textbtt{visit}, \textbtt{leave}.
      \item Se basa en la aplicación de un recorrido sobre los planes de evaluación resolviendo las dependencias para evaluar los atributos.
    \end{itemize}
  \end{block}

  \begin{exampleblock}{}
%     \begin{tabular}{p{3cm} c p{8cm}}
    \begin{center}
      \textbf{$\{ Eq_{1}, Eq_{2}, \ldots Eq_{n}\}$} $\Longrightarrow$  \textbf{\{visit($node_{1}$), compute($Eq_{1}$),$\ldots$,  compute($Eq_{n}$)\}}
    \end{center}
%     \end{tabular}
  \end{exampleblock}
}

\subsection*{Heurística del algoritmo}

\frame{
  \frametitle{Heurística del algoritmo}
  \begin{enumerate}
    \item Se recorre el plan tomando cada una de las ecuaciones.
\vspace{0.3cm}
    \item Dada la ecuación \textbtt{i}, en el plan, se debe computar todo su \textit{l\_value}. Para ello se deben resolver todas las dependencias dadas por su \textit{r\_value}.
\vspace{0.3cm}
    \item Dada una dependencia no resuelta en la ecuación, se analiza:
\vspace{0.3cm}
      \begin{itemize}
	\item \textbf{Símbolo de la parte izquierda de la regla} asociado a un \textbf{atributo heredado}, se realiza un ``\textbtt{leave}''.
\vspace{0.3cm}
	\item \textbf{Símbolo de la parte derecha de la regla} y además contiene un \textbf{atributo sintetizado}, se realiza un ``\textbtt{visit}''. \textbf{Selección del plan para el hijo}. 
      \end{itemize}
\vspace{0.3cm}
    \item Luego de resolver todas las dependencias, se realiza un \texttt{compute} del \textit{l\_value} y se marca a este como \textbtt{evaluado}.
  \end{enumerate}
}

\section{Usando magGen}

\frame{
    \frametitle{Usando \maggen}
    \begin{block}{}
     Luego de la instalación, \maggen\ puede ser invocado desde la línea de comandos.
    \end{block}

    \begin{exampleblock}{Argumentos de \maggen}
        \begin{description}
            \item [\textbtt{-f  file}] Entrada de \maggen\ como \textbtt{file}. Por defecto, (\textbtt{cin}) hasta \textbtt{EOF} (End Of File).

            \item [\textbtt{-i  header}] Incluir \textbtt{header} en la generación de código.

            \item [\textbtt{-fo folder}] Directorio de salida para \maggen. Por defecto ``\textit{./out\_maggen/}''.

            \item [\textbtt{-o  name}] Define a \textbtt{name} como el nombre de la clase y del archivo generado por \maggen. Por defecto el nombre ``\textbf{mag\_eval}''.

            \item [\textbtt{-h}] Muestra mensaje de ayuda.
        \end{description}
    \end{exampleblock}
}

\frame{
  \frametitle{Salida completa de \maggen}

  \begin{block}{}
      Al ejecutar \maggen\ se generan, en la carpeta de salida, lo siguiente:

      \begin{itemize}
        \item Archivos \textbf{Node.hpp}, \textbf{Plan.hpp}, $\langle \textbf{eval}\rangle.\textbf{cpp}$, $\langle \textbf{eval}\rangle.\textbf{hpp}$
        
        \item Archivo \textbf{grammar.log} con la gramática parseada.

        \item Carpeta \textbf{plan\_project} con los planes proyectados.

        \item Carpeta \textbf{plans}, con los planes calculados.

        \item Carpeta \textbf{graph}, que contiene subcarpetas con los tipos de grafos construidos.
      \end{itemize}
  \end{block}

  Tanto los planes como los grafos son generados en formato \texttt{dot}, que pueden ser convertidos a imágenes (png) utilizando el script brindado por la herramienta.
}

\frame{
    \frametitle{Usando \maggen}
    
    \begin{itemize}
        \item Se especifica en un archivo la MAG deseada.
        \vspace{0.3cm}

        \item Se invoca a \maggen\ con los parámetros adecuados desde la línea de comandos
        \vspace{0.3cm}
        \par{\underline{Ejemplo}: \textbtt{./maggen -f ./mag.input -fo ./out\_folder -o nameEval}}
        \vspace{0.3cm}

        \item Se construye el AST de entrada a evaluar en un archivo C++, junto con la invocación al \textbf{evaluador}.
        \vspace{0.3cm}

        \item Se compilan y se ejecuta el binario para lograr los resultados.
    \end{itemize}
    
}

\subsection*{Conclusiones}

\frame{
  \frametitle{Conclusiones}

\begin{itemize}
    \item Adquisición de conocimientos sobre las familias de GA, en particular MAG.
\vspace{0.3cm}
    \item Lenguaje de especificación para AG tiene sintaxis transparente, concisa y amplia.
\vspace{0.3cm}
    \item Construcción de secuencias de visitas mediante un algoritmo simple y eficiente que genera la mínima cantidad de secuencias visita.
\vspace{0.3cm}
    \item Los algoritmos de contrucción de planes y de secuencias de visitas mantienen propiedades de compatibilidad y consistencia con los grafos ADP.
\vspace{0.3cm}
    \item Construcción de planes sólo sobre las reglas alcanzables de la gramática.
\vspace{0.3cm}
    \item El evaluador generado es \textit{estático}, lo que garantiza que no hay procesamiento ``overhead'' en la evaluación.
    
%     \item La generación de código produce una representación compacta de cada estructura necesaria para la evaluación.

    \end{itemize}
}
